    <script>
        // CONFIGURAÇÃO SUPABASE
        const SUPABASE_URL = 'https://gfazevzzvgafmlnmkgjo.supabase.co';
        const SUPABASE_KEY = 'eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.eyJpc3MiOiJzdXBhYmFzZSIsInJlZiI6ImdmYXpldnp6dmdhZm1sbm1rZ2pvIiwicm9sZSI6ImFub24iLCJpYXQiOjE3NjU5NDE3MjMsImV4cCI6MjA4MTUxNzcyM30.00SUzPZP0-YCxwaynA0Eu8lBM2O7E4Sp52lwLVACRSo';

        const { createClient } = supabase;
        const supabaseClient = createClient(SUPABASE_URL, SUPABASE_KEY);

        // Pega o ID da URL
        const params = new URLSearchParams(window.location.search);
        const instrutorId = params.get('id');

        // Variável para guardar o preço real
        let valorAula = 0;

        // 1. Carrega os dados do instrutor ao abrir a tela
        async function carregarDados() {
            if (!instrutorId) {
                alert("Erro: Instrutor não identificado.");
                window.location.href = 'home.html';
                return;
            }

            const { data: instrutor, error } = await supabaseClient
                .from('instrutores')
                .select('*')
                .eq('id', instrutorId)
                .single();

            if (instrutor) {
                document.getElementById('txt-nome-instrutor').innerText = "Instrutor: " + instrutor.nome;
                document.getElementById('txt-endereco').innerText = instrutor.endereco;
                
                // Formata preço
                valorAula = instrutor.preco_aula;
                const precoFormatado = "R$ " + valorAula.toFixed(2).replace('.', ',');
                document.getElementById('txt-preco').innerText = precoFormatado;
                document.getElementById('txt-total').innerText = precoFormatado;
            }
        }

        // 2. Função de Salvar no Banco
        async function salvarAgendamento() {
            const btn = document.getElementById('btn-confirmar');
            
            // Muda o texto do botão para dar feedback visual
            btn.innerText = "Processando...";
            btn.disabled = true;

            // Cria o objeto para salvar
            // OBS: Como não temos sistema de login completo ainda, vamos salvar sem 'aluno_id'
            // ou usar um ID fictício se sua tabela exigir.
            const novoAgendamento = {
                instrutor_id: instrutorId,
                data_hora: new Date().toISOString(), // Salva a data de HOJE para teste
                status: 'confirmado',
                valor_aula: valorAula
            };

            const { data, error } = await supabaseClient
                .from('agendamentos')
                .insert([novoAgendamento]);

            if (error) {
                console.error(error);
                alert("Erro ao salvar agendamento. Tente novamente.");
                btn.innerText = "Tentar Novamente";
                btn.disabled = false;
            } else {
                // SUCESSO! Redireciona
                window.location.href = 'sucesso.html';
            }
        }

        // Inicia
        carregarDados();
    </script>
