        async function carregarDestaques() {
            const listaContainer = document.getElementById('lista-instrutores');
            
            // Busca apenas 5 instrutores para a Home (aleatórios ou os melhores avaliados)
            const { data: instrutores, error } = await supabaseClient
                .from('instrutores')
                .select('*')
                .eq('status', 'ativo')
                .limit(5);

            if (error) { console.error(error); return; }

            listaContainer.innerHTML = '';

            instrutores.forEach(instrutor => {
                const cardHTML = `
                <a href="detalhes.html?id=${instrutor.id}" class="block bg-slate-800 rounded-2xl overflow-hidden shadow-xl mb-6 relative group cursor-pointer hover:scale-[1.02] transition-transform duration-300">
                    <div class="h-48 relative">
                        <img src="${instrutor.foto_url}" class="w-full h-full object-cover" alt="${instrutor.nome}">
                        <div class="absolute inset-0 bg-gradient-to-t from-slate-900 via-transparent to-transparent"></div>
                        <div class="absolute top-4 right-4 bg-slate-900/90 px-3 py-1 rounded-lg flex items-center gap-1 backdrop-blur-sm">
                            <i class="fa-solid fa-star text-yellow-400 text-xs"></i>
                            <span class="text-sm font-bold">${instrutor.nota}</span>
                        </div>
                    </div>
                    <div class="absolute bottom-0 w-full p-4">
                        <h3 class="font-bold text-lg text-white">${instrutor.nome}</h3>
                        <p class="text-xs text-slate-300 mt-1">${instrutor.endereco}</p>
                    </div>
                </a>
                `;
                listaContainer.innerHTML += cardHTML;
            });
        }
