        window.onload = function() {
            const params = new URLSearchParams(window.location.search);
            const emailURL = params.get('email');
            const nomeURL = params.get('nome');
            const tipoURL = params.get('tipo');

            // 1. Aplica a seleção vinda da URL
            if(tipoURL) {
                selecionarTipo(tipoURL);
                
                // --- AQUI COMEÇA A BLINDAGEM ---
                const btnAluno = document.getElementById('btn-aluno');
                const btnInstrutor = document.getElementById('btn-instrutor');
                
                btnAluno.disabled = true;
                btnInstrutor.disabled = true;
                
                if(tipoURL === 'aluno') {
                    btnInstrutor.classList.add('opacity-30', 'cursor-not-allowed');
                    btnAluno.classList.add('cursor-not-allowed'); 
                } else {
                    btnAluno.classList.add('opacity-30', 'cursor-not-allowed');
                    btnInstrutor.classList.add('cursor-not-allowed');
                }

                document.getElementById('aviso-bloqueio').classList.remove('hidden');
                // --- FIM DA BLINDAGEM ---

                document.getElementById('email').value = emailURL;
                document.getElementById('senha').focus();
            }
            if(nomeURL) {
                document.getElementById('nome').value = nomeURL;
            }
        };

        function selecionarTipo(tipo) {
            tipoConta = tipo;
            const btnAluno = document.getElementById('btn-aluno');
            const btnInstrutor = document.getElementById('btn-instrutor');
            const areaInstrutor = document.getElementById('area-instrutor');

            if (tipo === 'aluno') {
                btnAluno.className = "border border-cyan-500 bg-cyan-500/20 text-cyan-400 py-2 rounded-lg font-bold transition-colors w-full";
                btnInstrutor.className = "border border-slate-600 text-slate-400 py-2 rounded-lg font-bold hover:bg-slate-700 transition-colors w-full";
                areaInstrutor.classList.add('hidden');
            } else {
                btnInstrutor.className = "border border-cyan-500 bg-cyan-500/20 text-cyan-400 py-2 rounded-lg font-bold transition-colors w-full";
                btnAluno.className = "border border-slate-600 text-slate-400 py-2 rounded-lg font-bold hover:bg-slate-700 transition-colors w-full";
                areaInstrutor.classList.remove('hidden');
            }
        }

        function getCheckedValues(name) {
            return Array.from(document.querySelectorAll(`input[name="${name}"]:checked`)).map(el => el.value);
        }

        async function cadastrar() {
            const nome = document.getElementById('nome').value;
            const email = document.getElementById('email').value;
            const senha = document.getElementById('senha').value;
            const btn = document.getElementById('btn-cadastrar');

            if (!nome || !email || !senha) { alert("Preencha todos os campos!"); return; }

            btn.innerHTML = '<i class="fa-solid fa-spinner animate-spin"></i> Criando...';
            btn.disabled = true;

            // 1. Cria Usuário
            const { data: user, error } = await supabaseClient
                .from('usuarios')
                .insert([{ nome, email, senha, tipo: tipoConta }])
                .select()
                .single();

            if (error) {
                alert("Erro ao criar usuário: " + error.message);
                btn.innerText = "Criar Conta";
                btn.disabled = false;
                return;
            }

            // 2. Se for INSTRUTOR, cria perfil (AGORA COM VERIFICAÇÃO DE ERRO)
            if (tipoConta === 'instrutor') {
                const categorias = getCheckedValues('cat');
                const cambios = getCheckedValues('cambio');
                const servicos = getCheckedValues('servico');
                const aceitaVeiculoProprio = document.getElementById('veiculo_proprio').checked;

                console.log("Tentando criar perfil de instrutor para ID:", user.id);

                const { data: perfil, error: erroInstrutor } = await supabaseClient.from('instrutores').insert([{
                    nome: nome,
                    status: 'pendente',
                    preco_aula: 80.00,
                    foto_url: 'https://cdn-icons-png.flaticon.com/512/3135/3135715.png',
                    endereco: 'Localização a definir',
                    categoria: 'Autônomo',
                    categorias: categorias.length ? categorias : ['B'],
                    tipos_servico: servicos,
                    cambio: cambios,
                    atende_veiculo_proprio: aceitaVeiculoProprio,
                    nota: 5.0,
                    avaliacoes_count: 0,
                    usuario_id: user.id
                }]).select();

                // SE DER ERRO AQUI, É PORQUE FALTAM COLUNAS NO BANCO
                if (erroInstrutor) {
                    console.error("Erro Supabase:", erroInstrutor);
                    alert("ERRO CRÍTICO: " + erroInstrutor.message + "\n\nO banco de dados rejeitou o cadastro. Verifique se as colunas (usuario_id, categorias, etc) foram criadas.");
                    
                    // SEGURANÇA: Apaga o usuário criado para não travar o email
                    await supabaseClient.from('usuarios').delete().eq('id', user.id);
                    
                    btn.innerText = "Tentar Novamente";
                    btn.disabled = false;
                    return;
                }
            }

            // 3. Sucesso (Só chega aqui se tudo deu certo)
            localStorage.setItem('usuario_logado', JSON.stringify(user));
            alert("Cadastro realizado com sucesso!");
            
            if(tipoConta === 'instrutor') {
                window.location.href = 'aguardando_aprovacao.html'; 
            } else {
                window.location.href = 'home.html';
            }
        }
